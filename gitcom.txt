

Контрольная сумма(hash) 	некоторое значение, рассчитанное с применением определённого алгоритма и 						используемое для проверки целостности данных при их передаче или хранении 						(аналогия - идентификационный код товара).

X: 				сделать текущим диском диск X.

cd /(каталог) 			вход в папку.

git init 			создаёт в текущей директории новую поддиректорию с именем .git, содержащую всю 						структуру Git-репозитория.

mkdir (каталог) 		создает новый каталог.

ls				просмотр файлов в дириктории.

		*КОНФИГУРАЦИЯ*
git config 			просмотр и настройка параметров Git.
		Уровни конфигурации
--system			установление настроек Git в общей операционной системе.	^(настройки)
--global			установление настроек Git на уровне пользователя(часто)	|(ищет)
--local(default)		установление настроек Git на уровне проекта.		|(Git)

git config user.name '' 	задать имя пользователя(без имени нек. команды не раб.).	(в проекте)
git config user.email f@ff	задать mail пользователя(без mail нек. команды не раб.).	(в проекте)
git config --global user.name ''						на уровне пользователя(часто)
git config --global user.email f@ff						на уровне пользователя(часто)
git config --unset user.name		удалить имя						(в проекте)
git config --remove-section user	удаление всей секции user				(в проекте)	
cat .git/config			просмотр имя и mail пользователя.	
git config (--global) --list	просмотр параметров конфигурации.	(на уровне пользователя)

		*ОСНОВЫ*
git add		добавите файлов в индекс (chaged area) под версионный контроль.
